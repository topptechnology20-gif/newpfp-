function generatePumpFunFighterProfile(coin) {
  const ageInMinutes = (Date.now() / 1000 - coin.created_timestamp) / 60;
  const ageInHours   = ageInMinutes / 60;

  return {
    // Identity
    source:        'PUMPFUN',
    source_id:     coin.mint,
    display_name:  `${coin.symbol}`,
    full_name:     coin.name,
    image_uri:     coin.image_uri,   // raw — processed in Part 3
    community_tag: coin.symbol,

    // Combat Profile traits (0-100)
    aggression:   normalize(coin.usd_market_cap, 0, 1_000_000) * 100,
    // Higher mcap = more aggressive (established force)

    defense:      normalize(coin.total_supply / 1e9, 0, 10) * 100,
    // More tokens = more resilient (harder to move)

    intelligence: normalize(ageInHours, 0, 72) * 100,
    // Older coin = smarter (survived longer, not a 5-min rug)

    speed:        normalize(coin.volume, 0, 500_000) * 100,
    // Volume = speed (how fast money is moving through it)

    luck:         coin.king_of_the_hill_timestamp ? 85 : normalize(Math.random(), 0, 1) * 40 + 20,
    // Was it ever King of the Hill? Lucky. Otherwise mild random.

    hp: 80 + normalize(coin.usd_market_cap, 0, 500_000) * 120,
    // HP 80-200 based on market cap

    // Arena metadata
    effective_tier: 'none',    // no tools equipped yet — seeded as Tier None (bait fighters)
    is_pumpfun_seed: true,
    is_king_of_hill: !!coin.king_of_the_hill_timestamp,
    graduated:       coin.complete,
  };
}

This is the most important part for the Avatar Mode integration. The logo goes through three steps:

javascript
// server/pumpfun/imagePipeline.js
// Uses: sharp (npm install sharp) + node-fetch

const sharp = require('sharp');
const fetch = require('node-fetch');

async function processCoinLogo(imageUri) {
  try {
    // Step 1: Fetch the raw image
    // pump.fun uses IPFS - two fallback gateways
    const url = imageUri
      .replace('https://cf-ipfs.com/ipfs/', 'https://ipfs.io/ipfs/')

    const res = await fetch(url, { timeout: 8000 });
    const buffer = await res.buffer();

    // Step 2: Resize + circular crop
    const size = 200;
    const circleOverlay = Buffer.from(
      `<svg><circle cx="${size/2}" cy="${size/2}" r="${size/2}"/></svg>`
    );

    const processed = await sharp(buffer)
      .resize(size, size, { fit: 'cover', position: 'centre' })
      .png()
      .composite([{ input: circleOverlay, blend: 'dest-in' }])
      // Step 3: Normalize brightness (prevents black squares)
      .modulate({ brightness: 1.05, saturation: 1.1 })
      .toBuffer();

    // Step 4: Store (Replit = write to /tmp or upload to Cloudinary/S3)
    // For Replit dev: return base64 data URI
    const base64 = processed.toString('base64');
    return `data:image/png;base64,${base64}`;

  } catch (err) {
    console.error('Image pipeline failed for', imageUri, err.message);
    return null; // fighter gets Bantah default head as fallback
  }
}

const REFRESH_INTERVAL = 5 * 60 * 1000; // 5 minutes
const MAX_PUMPFUN_FIGHTERS_IN_ARENA = 20;

async function seedArenaFromPumpFun() {
  console.log('[PumpFun Seeder] Fetching trending coins...');

  // Fetch three pools
  const [trending, recent, kotH] = await Promise.all([
    fetchTrendingCoins(15),
    fetchRecentlyTraded(5),
    fetchKingOfHill(),
  ]);

  // Merge + deduplicate by mint
  const allCoins = dedupeByMint([...trending, ...recent, kotH]);

  // Filter valid
  const validCoins = allCoins.filter(isValidForArena);

  // Remove stale pump.fun fighters from arena
  // (coins that are no longer trending get replaced)
  await db.execute(
    `UPDATE bota_fighter_profiles
     SET arena_status = 'retired'
     WHERE source = 'PUMPFUN'
     AND arena_status = 'active'`
  );

  // Seed new fighters
  for (const coin of validCoins.slice(0, MAX_PUMPFUN_FIGHTERS_IN_ARENA)) {
    const profile = generatePumpFunFighterProfile(coin);
    const avatar  = await processCoinLogo(coin.image_uri);

    await upsertFighter({
      ...profile,
      processed_avatar: avatar || null,
      arena_status: 'active',
      admin_tier: coin.king_of_the_hill_timestamp ? 'D_LEGENDARY' : 'B_STANDARD',
      // KotH coin = Legendary seed agent in the arena
    });
  }

  console.log(`[PumpFun Seeder] ${validCoins.length} fighters seeded.`);
}

// Run on server start + every 5 minutes
seedArenaFromPumpFun();
setInterval(seedArenaFromPumpFun, REFRESH_INTERVAL);

function normalizeDexToken(pair, chainId, iconUrl) {
  const ageSeconds = pair.pairCreatedAt
    ? (Date.now() - pair.pairCreatedAt) / 1000
    : 86400;                         // default 1 day if unknown

  return {
    unique_id:      `${chainId}_${pair.baseToken.address}`,
    chain:          chainId,
    mint_address:   pair.baseToken.address,
    display_name:   pair.baseToken.symbol,
    full_name:      pair.baseToken.name,
    image_uri:      iconUrl || pair.info?.imageUrl || null,

    market_cap_usd: pair.marketCap || 0,
    volume_24h_usd: pair.volume?.h24 || 0,
    holder_count:   null,            // not in pair data
    age_seconds:    ageSeconds,
    price_usd:      parseFloat(pair.priceUsd) || 0,
    is_king:        false,           // chain-specific concept
    graduated:      true,            // DEX-listed = graduated
  };

function normalize(val, min, max) {
  return Math.min(100, Math.max(0, ((val - min) / (max - min)) * 100));
}

export function generateCrossChainFighterProfile(coin) {
  const ageHours = coin.age_seconds / 3600;

  const traits = {
    aggression:   normalize(coin.market_cap_usd,  0, 5_000_000) * 100,
    defense:      normalize(ageHours,              0, 168) * 100,  // 1 week = max defense
    intelligence: normalize(ageHours,              0, 72)  * 100,  // survival = smart
    speed:        normalize(coin.volume_24h_usd,   0, 2_000_000) * 100,
    luck:         coin.is_king ? 88 : normalize(Math.random(), 0, 1) * 50 + 20,
  };

  const hp = 80 + normalize(coin.market_cap_usd, 0, 5_000_000) * 120;

  // Chain personality modifiers
  // Each chain has a meta flavor baked in
  const chainModifiers = {
    solana: { speed: +15, luck: +10 },       // Solana = fast + volatile
    bsc:    { defense: +10, aggression: +8 }, // BSC = older, more established memes
    base:   { intelligence: +12 },            // Base = newer, builder community
  };

  const mods = chainModifiers[coin.chain] || {};
  Object.keys(mods).forEach(k => {
    traits[k] = Math.min(100, traits[k] + mods[k]);
  });

  return {
    unique_id:       coin.unique_id,
    chain:           coin.chain,
    source:          'TRENDING',
    display_name:    coin.display_name,
    full_name:       coin.full_name,
    image_uri:       coin.image_uri,
    processed_avatar: null,          // filled by image pipeline

    // Combat profile
    aggression:      Math.round(traits.aggression),
    defense:         Math.round(traits.defense),
    intelligence:    Math.round(traits.intelligence),
    speed:           Math.round(traits.speed),
    luck:            Math.round(traits.luck),
    hp:              Math.round(hp),

    // Arena metadata
    arena_status:   'active',
    effective_tier: 'none',
    is_king:        coin.is_king,
    admin_tier:     coin.is_king ? 'D_LEGENDARY' : 'B_STANDARD',

    // Source metrics (stored for leaderboard display)
    market_cap_usd: coin.market_cap_usd,
    volume_24h_usd: coin.volume_24h_usd,
    last_refreshed: new Date().toISOString(),
  };
}
MATCHING ENGINE
The most important part — how fighters get paired:

javascript
// server/arena/matchingEngine.js

// ── ECPS formula ─────────────────────────────────────────────────
function calculateECPS(fighter) {
  return (
    fighter.aggression   * 0.30 +
    fighter.defense      * 0.20 +
    fighter.speed        * 0.20 +
    fighter.intelligence * 0.20 +
    fighter.luck         * 0.10
  );
}

// ── Match type priorities ─────────────────────────────────────────
/*
  1. CHAIN RIVALRY    — SOL vs BSC, BSC vs BASE, SOL vs BASE (most drama)
  2. KING CLASH       — any two is_king fighters (premium event)
  3. ECPS-BALANCED    — same chain, similar power score
  4. RANDOM FILL      — anything vs anything (fallback if pool is thin)
*/

export function buildMatchQueue(activeFighters) {
  const queue = [];
  const available = [...activeFighters];   // copy so we can splice
  const matched   = new Set();

  // ── Pass 1: King clashes ─────────────────────────────────────
  const kings = available.filter(f => f.is_king);
  if (kings.length >= 2) {
    for (let i = 0; i < kings.length - 1; i += 2) {
      queue.push(createMatch(kings[i], kings[i+1], 'KING_CLASH'));
      matched.add(kings[i].unique_id);
      matched.add(kings[i+1].unique_id);
    }
  }

  const remaining = available.filter(f => !matched.has(f.unique_id));

  // ── Pass 2: Cross-chain rivalry pairs ────────────────────────
  const byChain = {
    solana: remaining.filter(f => f.chain === 'solana'),
    bsc:    remaining.filter(f => f.chain === 'bsc'),
    base:   remaining.filter(f => f.chain === 'base'),
  };

  const rivalryPairs = [
    [byChain.solana, byChain.bsc,  'SOL_VS_BSC'],
    [byChain.bsc,   byChain.base,  'BSC_VS_BASE'],
    [byChain.solana, byChain.base, 'SOL_VS_BASE'],
  ];

  for (const [poolA, poolB, matchType] of rivalryPairs) {
    const unusedA = poolA.filter(f => !matched.has(f.unique_id));
    const unusedB = poolB.filter(f => !matched.has(f.unique_id));

    // Pair closest ECPS across chains
    for (const a of unusedA) {
      if (matched.has(a.unique_id)) continue;
      const ecpsA = calculateECPS(a);

      const opponent = unusedB
        .filter(b => !matched.has(b.unique_id))
        .sort((x, y) =>
          Math.abs(calculateECPS(x) - ecpsA) -
          Math.abs(calculateECPS(y) - ecpsA)
        )[0];

      if (!opponent) continue;

      const ecpsDiff = Math.abs(calculateECPS(opponent) - ecpsA);
      if (ecpsDiff / ecpsA <= 0.25) {    // within 25% for cross-chain
        queue.push(createMatch(a, opponent, matchType));
        matched.add(a.unique_id);
        matched.add(opponent.unique_id);
      }
    }
  }

  // ── Pass 3: Same-chain ECPS balanced ─────────────────────────
  const leftover = remaining.filter(f => !matched.has(f.unique_id));

  for (let i = 0; i < leftover.length; i++) {
    const a = leftover[i];
    if (matched.has(a.unique_id)) continue;
    const ecpsA = calculateECPS(a);

    const opponent = leftover
      .filter((b, j) => j !== i && !matched.has(b.unique_id))
      .sort((x, y) =>
        Math.abs(calculateECPS(x) - ecpsA) -
        Math.abs(calculateECPS(y) - ecpsA)
      )[0];

    if (!opponent) continue;

    queue.push(createMatch(a, opponent, 'BALANCED'));
    matched.add(a.unique_id);
    matched.add(opponent.unique_id);
  }

  return queue;
}

function createMatch(fighterA, fighterB, matchType) {
  return {
    id:          `match_${Date.now()}_${Math.random().toString(36).slice(2,7)}`,
    fighter_a:   fighterA,
    fighter_b:   fighterB,
    match_type:  matchType,
    created_at:  new Date().toISOString(),
    status:      'pending',

    // Display label for UI
    label: matchTypeLabel(matchType, fighterA, fighterB),
  };
}

function matchTypeLabel(type, a, b) {
  const labels = {
    KING_CLASH:  `👑 King Clash — ${a.display_name} vs ${b.display_name}`,
    SOL_VS_BSC:  `⚔️ Chain Rivalry — SOL vs BSC`,
    BSC_VS_BASE: `⚔️ Chain Rivalry — BSC vs BASE`,
    SOL_VS_BASE: `⚔️ Chain Rivalry — SOL vs BASE`,
    BALANCED:    `🥊 ${a.display_name} vs ${b.display_name}`,
  };
  return labels[type] || `${a.display_name} vs ${b.display_name}`;
}

// Called immediately after every battle resolves
export async function updateLeaderboard(battleResult) {
  const { winner, loser, match_type, bc_earned, usdt_earned } = battleResult;

  // ── Update winner ────────────────────────────────────────────
  await db.execute(`
    UPDATE trending_fighter_stats SET
      wins            = wins + 1,
      total_bc_earned = total_bc_earned + $1,
      win_streak      = win_streak + 1,
      last_fought_at  = NOW()
    WHERE fighter_unique_id = $2
  `, [bc_earned, winner.unique_id]);

  // ── Update loser ─────────────────────────────────────────────
  await db.execute(`
    UPDATE trending_fighter_stats SET
      losses          = losses + 1,
      win_streak      = 0,
      last_fought_at  = NOW()
    WHERE fighter_unique_id = $1
  `, [loser.unique_id]);

  // ── Recalculate global rank ───────────────────────────────────
  // Rank = weighted: 50% wins, 30% win_streak, 20% bc_earned
  await db.execute(`
    UPDATE trending_fighter_stats SET
      global_rank = sub.rank
    FROM (
      SELECT fighter_unique_id,
             RANK() OVER (
               ORDER BY (wins * 0.5 + win_streak * 0.3 + total_bc_earned * 0.2) DESC
             ) AS rank
      FROM trending_fighter_stats
    ) sub
    WHERE trending_fighter_stats.fighter_unique_id = sub.fighter_unique_id
  `);

  // ── Recalculate per-chain rank ────────────────────────────────
  for (const chain of ['solana', 'bsc', 'base']) {
    await db.execute(`
      UPDATE trending_fighter_stats SET
        chain_rank = sub.rank
      FROM (
        SELECT fighter_unique_id,
               RANK() OVER (ORDER BY wins DESC) AS rank
        FROM trending_fighter_stats
        WHERE chain = $1
      ) sub
      WHERE trending_fighter_stats.fighter_unique_id = sub.fighter_unique_id
      AND trending_fighter_stats.chain = $1
    `, [chain]);
  }

  // ── Cross-chain champion badge ────────────────────────────────
  // If winner has won against 2+ different chains, gets cross-chain badge
  const distinctChainsBeaten = await db.query(`
    SELECT COUNT(DISTINCT loser_chain) as chains
    FROM battle_results
    WHERE winner_id = $1
  `, [winner.unique_id]);

  if (distinctChainsBeaten[0].chains >= 2) {
    await db.execute(`
      UPDATE trending_fighter_stats
      SET badges = array_append(badges, 'CHAIN_BREAKER')
      WHERE fighter_unique_id = $1
      AND NOT ('CHAIN_BREAKER' = ANY(badges))
    `, [winner.unique_id]);
  }
}

export async function getTrendingFighterList({ chain, sort, status, limit = 30 }) {
  const chainFilter  = chain  !== 'all' ? `AND f.chain = '${chain}'`   : '';
  const statusFilter = status !== 'all' ? `AND f.arena_status = '${status}'` : '';

  const sortOptions = {
    rank:       's.global_rank ASC',
    mcap:       'f.market_cap_usd DESC',
    wins:       's.wins DESC',
    streak:     's.win_streak DESC',
    recent:     's.last_fought_at DESC',
  };
  const orderBy = sortOptions[sort] || sortOptions.rank;

  return db.query(`
    SELECT
      f.unique_id,
      f.chain,
      f.display_name,
      f.processed_avatar,
      f.arena_status,
      f.is_king,
      f.admin_tier,
      f.market_cap_usd,
      f.aggression,
      f.defense,
      f.intelligence,
      f.speed,
      f.luck,
      f.hp,
      s.wins,
      s.losses,
      s.win_streak,
      s.global_rank,
      s.chain_rank,
      s.badges
    FROM trending_fighter_profiles f
    LEFT JOIN trending_fighter_stats s ON s.fighter_unique_id = f.unique_id
    WHERE 1=1 ${chainFilter} ${statusFilter}
    ORDER BY ${orderBy}
    LIMIT $1
  `, [limit]);
}

javascript
// server/arena/trendingOrchestrator.js

import { fetchSolanaTrending, fetchBSCTrending, fetchBaseTrending } from '../chains/fetchers.js';
import { generateCrossChainFighterProfile } from '../chains/fighterGenerator.js';
import { processCoinLogo }                   from '../chains/imagePipeline.js';
import { buildMatchQueue }                   from './matchingEngine.js';
import { runBattle }                          from './battleEngine.js';
import { updateLeaderboard }                  from './leaderboardService.js';

const REFRESH_MS = 5 * 60 * 1000;

async function refreshCycle() {
  console.log('[Trending] Refresh cycle started');

  // 1. Fetch all chains in parallel
  const [solana, bsc, base] = await Promise.allSettled([
    fetchSolanaTrending(6),
    fetchBSCTrending(5),
    fetchBaseTrending(5),
  ]);

  const allCoins = [
    ...(solana.status === 'fulfilled' ? solana.value : []),
    ...(bsc.status    === 'fulfilled' ? bsc.value    : []),
    ...(base.status   === 'fulfilled' ? base.value   : []),
  ];

  // 2. Generate + upsert fighter profiles
  const fighters = [];
  for (const coin of allCoins) {
    const profile = generateCrossChainFighterProfile(coin);
    const avatar  = await processCoinLogo(coin.image_uri);
    profile.processed_avatar = avatar;
    await upsertFighterProfile(profile);
    fighters.push(profile);
  }

  // 3. Build match queue
  const matchQueue = buildMatchQueue(fighters);
  console.log(`[Trending] ${matchQueue.length} matches queued`);

  // 4. Simulate all matches
  for (const match of matchQueue) {
    const result = await runBattle(match.fighter_a, match.fighter_b, match.match_type);
    await updateLeaderboard(result);
    await saveBattleResult(result);

    // Emit real-time update to connected clients
    io.emit('battle:result', {
      match_type:  match.match_type,
      label:       match.label,
      winner_name: result.winner.display_name,
      winner_chain: result.winner.chain,
      loser_name:  result.loser.display_name,
      loser_chain: result.loser.chain,
      rounds:      result.rounds_fought,
    });
  }

  // 5. Emit leaderboard update signal
  io.emit('leaderboard:updated');

  console.log('[Trending] Refresh cycle complete');
}

// Boot + schedule
refreshCycle();
setInterval(refreshCycle, REFRESH_MS);

HOW LEADERBOARD + FIGHTER LIST STAY LIVE
Socket.io emits 'leaderboard:updated' after every refresh cycle
        ↓
Frontend listens, calls GET /api/leaderboard/trending
        ↓
Leaderboard re-renders with new ranks, streaks, badges

Same socket emits 'battle:result' per fight as it completes
        ↓
Live battle feed at the bottom of both pages updates
        ↓
Chain tabs on fighter list show updated win/loss counts
        ↓
Users watching see real-time ranking shifts
